    <script>
        // Global variables
        let userSubmissions = [];
        let capturedPhotos = [];
        let stream = null;

        // FIXED: Proper form submission handling
        document.addEventListener("DOMContentLoaded", function () {
            console.log("üü¢ Page loaded successfully");
            
            const form = document.getElementById("listingForm");
            if (!form) {
                console.error("‚ùå Form with id 'listingForm' not found!");
                return;
            }
            
            console.log("‚úÖ Form found, adding event listener");
            
            form.addEventListener("submit", function (event) {
                console.log("üöÄ Form submission triggered");
                event.preventDefault(); // Prevent default form submission
                
                // Show loading state
                const submitButton = form.querySelector('button[type="submit"]');
                const originalText = submitButton.textContent;
                submitButton.textContent = 'Submitting...';
                submitButton.disabled = true;
                
                try {
                    const formData = new FormData(form);
                    
                    // Debug: Log form data
                    console.log("üìã Form data being sent:");
                    for (let [key, value] of formData.entries()) {
                        if (value instanceof File) {
                            console.log(`  ${key}: File "${value.name}" (${value.size} bytes)`);
                        } else {
                            console.log(`  ${key}: "${value}"`);
                        }
                    }
                    
                    // Add multiple files properly
                    const files = document.getElementById('fileInput').files;
                    console.log(`üìÅ Adding ${files.length} files`);
                    
                    // Clear existing fileInput entries and add all files
                    formData.delete('fileInput');
                    for (let i = 0; i < files.length; i++) {
                        formData.append("fileInput", files[i]);
                    }
                    
                    console.log("üåê Sending POST request to /submit_listing");

                    fetch("/submit_listing", {
                        method: "POST",
                        body: formData
                    })
                    .then(response => {
                        console.log(`üì° Response status: ${response.status}`);
                        return response.json();
                    })
                    .then(data => {
                        console.log("üì¶ Response data:", data);
                        
                        if (data.success) {
                            console.log("‚úÖ Success!");
                            showMessage("‚úÖ Success! Your listing was submitted.", 'success');
                            form.reset();
                            updatePhotoCount();
                        } else {
                            console.log("‚ùå Server error:", data.message);
                            showMessage("‚ùå Failed: " + data.message, 'error');
                        }
                    })
                    .catch(error => {
                        console.error("üí• Network error:", error);
                        showMessage("‚ùå Network error occurred during submission.", 'error');
                    })
                    .finally(() => {
                        // Restore button state
                        submitButton.textContent = originalText;
                        submitButton.disabled = false;
                    });
                    
                } catch (error) {
                    console.error("üí• JavaScript error:", error);
                    showMessage("‚ùå Error occurred during submission.", 'error');
                    
                    // Restore button state
                    submitButton.textContent = originalText;
                    submitButton.disabled = false;
                }
            });
        });

        // Show message function
        function showMessage(message, type) {
            const messageContainer = document.getElementById('messageContainer');
            messageContainer.innerHTML = `<div class="${type}-message">${message}</div>`;
            
            // Clear message after 5 seconds
            setTimeout(() => {
                messageContainer.innerHTML = '';
            }, 5000);
        }

        // Category selection
        function selectCategory(category) {
            // Remove selection from all cards
            document.querySelectorAll('.category-card').forEach(card => {
                card.classList.remove('selected');
            });
            
            // Add selection to clicked card
            event.target.closest('.category-card').classList.add('selected');
            
            // Pre-select the category in the form
            document.getElementById('materialType').value = category;
        }

        // Navigation functions
        function showSellForm() {
            showTab('sell');
            document.querySelector('.tab[onclick="showTab(\'sell\')"]').scrollIntoView({ behavior: 'smooth' });
        }

        function showTab(tabName) {
            // Hide all tab contents
            document.querySelectorAll('.tab-content').forEach(content => {
                content.classList.add('hidden');
            });
            
            // Remove active class from all tabs
            document.querySelectorAll('.tab').forEach(tab => {
                tab.classList.remove('active');
            });
            
            // Show selected tab content
            document.getElementById(tabName + '-content').classList.remove('hidden');
            
            // Add active class to clicked tab
            event.target.classList.add('active');
        }

        // Camera and photo functions
        function updatePhotoCount() {
            const fileInput = document.getElementById('fileInput');
            const photoCount = document.getElementById('photoCount');
            const count = fileInput.files.length + capturedPhotos.length;
            photoCount.textContent = `${count}/5`;
        }

        function handleFileSelect(event) {
            console.log("üìÅ Files selected:", event.target.files.length);
            displayPhotos();
            updatePhotoCount();
        }

        function displayPhotos() {
            const preview = document.getElementById('photoPreview');
            preview.innerHTML = '';
            
            // Display uploaded files
            const files = document.getElementById('fileInput').files;
            Array.from(files).forEach((file, index) => {
                const reader = new FileReader();
                reader.onload = function(e) {
                    const photoItem = document.createElement('div');
                    photoItem.className = 'photo-item';
                    photoItem.innerHTML = `
                        <img src="${e.target.result}" alt="Upload ${index + 1}">
                        <button type="button" class="photo-remove" onclick="removeUploadedPhoto(${index})">&times;</button>
                    `;
                    preview.appendChild(photoItem);
                };
                reader.readAsDataURL(file);
            });
            
            // Display captured photos
            capturedPhotos.forEach((photo, index) => {
                const photoItem = document.createElement('div');
                photoItem.className = 'photo-item';
                photoItem.innerHTML = `
                    <img src="${photo}" alt="Captured ${index + 1}">
                    <button type="button" class="photo-remove" onclick="removeCapturedPhoto(${index})">&times;</button>
                `;
                preview.appendChild(photoItem);
            });
        }

        function removeUploadedPhoto(index) {
            const fileInput = document.getElementById('fileInput');
            const files = Array.from(fileInput.files);
            files.splice(index, 1);
            
            // Create new FileList
            const dt = new DataTransfer();
            files.forEach(file => dt.items.add(file));
            fileInput.files = dt.files;
            
            displayPhotos();
            updatePhotoCount();
        }

        function removeCapturedPhoto(index) {
            capturedPhotos.splice(index, 1);
            displayPhotos();
            updatePhotoCount();
        }

        async function startCamera() {
            try {
                stream = await navigator.mediaDevices.getUserMedia({ 
                    video: { facingMode: 'environment' } 
                });
                
                const video = document.getElementById('cameraVideo');
                const controls = document.getElementById('captureControls');
                
                video.srcObject = stream;
                video.classList.remove('hidden');
                controls.classList.remove('hidden');
                
                console.log("üì∑ Camera started");
            } catch (error) {
                console.error("‚ùå Camera error:", error);
                alert("Could not access camera. Please use the upload option instead.");
            }
        }

        function capturePhoto() {
            const video = document.getElementById('cameraVideo');
            const canvas = document.getElementById('cameraCanvas');
            const context = canvas.getContext('2d');
            
            canvas.width = video.videoWidth;
            canvas.height = video.videoHeight;
            context.drawImage(video, 0, 0);
            
            const photoData = canvas.toDataURL('image/jpeg', 0.8);
            capturedPhotos.push(photoData);
            
            console.log("üì∏ Photo captured");
            displayPhotos();
            updatePhotoCount();
            stopCamera();
        }

        function stopCamera() {
            if (stream) {
                stream.getTracks().forEach(track => track.stop());
                stream = null;
            }
            
            const video = document.getElementById('cameraVideo');
            const controls = document.getElementById('captureControls');
            
            video.classList.add('hidden');
            controls.classList.add('hidden');
            
            console.log("üì∑ Camera stopped");
        }

        // Initialize on page load
        document.addEventListener('DOMContentLoaded', function() {
            updatePhotoCount();
        });
    </script>
